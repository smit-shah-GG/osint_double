  1. Add `ExtractionMetrics` dataclass at module level (after PipelineStats):
     ```python
     @dataclass
     class ExtractionMetrics:
         """Per-model extraction success/failure tracking."""
         model_id: str
         success_count: int = 0
         failure_count: int = 0
         repair_count: int = 0
         total_facts: int = 0
         total_duration_ms: float = 0.0

         @property
         def success_rate(self) -> float:
             total = self.success_count + self.failure_count
             return self.success_count / total if total > 0 else 0.0
     ```

  2. Add `_model_metrics: dict[str, ExtractionMetrics]` to ExtractionPipeline.__init__.

  3. Modify the `_extract_one` inner function in `process_investigation` to track per-article stats:
     ```python
     async def _extract_one(article: Dict[str, Any]) -> Optional[List[Dict[str, Any]]]:
         nonlocal completed
         async with sem:
             start = time.monotonic()
             content = self._article_to_content(article, investigation_id)
             result = await self._extract_with_error_handling(content)
             elapsed_ms = (time.monotonic() - start) * 1000
             completed += 1

             # Per-article logging
             url = article.get("url", "unknown")[:80]
             fact_count = len(result) if result else 0
             self.logger.info(
                 "article_extracted",
                 url=url,
                 facts=fact_count,
                 duration_ms=round(elapsed_ms, 1),
                 progress=f"{completed}/{len(articles)}",
             )

             # Update model metrics
             model_id = self.extraction_agent.model_name
             if model_id not in self._model_metrics:
                 self._model_metrics[model_id] = ExtractionMetrics(model_id=model_id)
             metrics = self._model_metrics[model_id]
             if result is not None:
                 metrics.success_count += 1
                 metrics.total_facts += fact_count
             else:
                 metrics.failure_count += 1
             metrics.total_duration_ms += elapsed_ms

             return result
     ```

     Add `import time` at module top.

     import asyncio

     # Warn-once for fallback transitions: set of (primary, fallback) tuples
     _warned_transitions: set[tuple[str, str]] = set()
     _warned_lock: asyncio.Lock | None = None

     def _get_warned_lock() -> asyncio.Lock:
         global _warned_lock
         if _warned_lock is None:
